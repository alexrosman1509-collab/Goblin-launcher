        function getUpgradeCost(type, currentLevel) {
            if (type === 'power') return Math.floor(4 * Math.pow(1.08, currentLevel - 1));
            if (type === 'earnings') return Math.floor(5 * Math.pow(1.09, currentLevel - 1));
            if (type === 'offline') return Math.floor(6 * Math.pow(1.10, currentLevel));
            return 9999;
        }

        function processOfflineEarnings() {
            if (profile.upgrades.offline > 0) {
                let now = Date.now();
                let deltaMs = now - profile.lastSavedTimestamp;
                let minutesPassed = deltaMs / 60000;
                let calculatedYield = Math.floor(minutesPassed * (profile.upgrades.offline * 2.0));
                if (calculatedYield > 0) {
                    profile.goldVault += calculatedYield;
                    alert(`Your goblin clansmen collected ${calculatedYield} gold while you were away!`);
                    saveProfile();
                }
            }
        }

        function buyUpgrade(type) {

                profile.goldVault -= cost;
                profile.upgrades[type]++;
                saveProfile();
                updateMenuShopDOM();
            }
        }

        function triggerWheelSpin(freeSpinToken = false) {
            if (!freeSpinToken && profile.gemVault < 2 || wheelIsSpinning) return;
            if (!freeSpinToken) profile.gemVault -= 2;
            
            isFreeSpin = freeSpinToken;
            wheelIsSpinning = true;

        }

        function processWheelPhysicsTick() {
            if (!wheelIsSpinning && wheelSpeed === 0) return;

            wheelAngle += wheelSpeed;
            wheelSpeed *= 0.983; 

            let sectorArc = (Math.PI * 2) / 16;
            let currentPegSector = Math.floor(wheelAngle / sectorArc);
            let priorPegSector = Math.floor((wheelAngle - wheelSpeed) / sectorArc);
            
            if (currentPegSector !== priorPegSector) {
                pinBunkAnimationCounter = 1.0; 
            }

            if (wheelSpeed < 0.0015) {
                wheelSpeed = 0;
                wheelIsSpinning = false;
                evaluateWheelStopPrize();
            }

            drawSpinningWheelContext();
        }

        function evaluateWheelStopPrize() {
            let normalizedAngle = (Math.PI * 1.5 - wheelAngle) % (Math.PI * 2);
            if (normalizedAngle < 0) normalizedAngle += Math.PI * 2;

            let segmentArc = (Math.PI * 2) / 16;
            let prizeIndex = Math.floor(normalizedAngle / segmentArc) % 16;
            let prize = WHEEL_PRIZES[prizeIndex];
            finalPrizeObject = prize;

            let outMsg = "";

            if (prize.type === 'stash') {
                profile.gemVault += 15;
                outMsg = `JACKPOT! +15 Emeralds Added!`;
            } else if (prize.type === 'time_warp') {
                let timeWarpYield = Math.floor(240 * (profile.upgrades.offline * 2.0)); 
                profile.goldVault += timeWarpYield;
                outMsg = `TIME WARP! +${timeWarpYield} Gold (4H Yield)!`;
            } else if (prize.type === 'boost') {
                let endAttr = prize.id + 'End';
                profile.boosts[endAttr] = Math.max(Date.now(), profile.boosts[endAttr]) + prize.val;
                outMsg = `Buff Activated: ${prize.text}!`;
            } else if (prize.type === 'nothing') {
                outMsg = "Unlucky! You spun absolutely nothing.";
            } else if (prize.type === 'extra_spin') {
                outMsg = "🎫 LUCKY ROLL! Free extra spin earned! Click claim to roll again!";
            } else if (prize.type === 'goblin') {
                if (profile.unlockedGoblins.includes(prize.id)) {
                    profile.goldVault += 100;
                    outMsg = `Skin Duplicate! Converted directly into +100 Gold Coins!`;
                } else {
                    profile.unlockedGoblins.push(prize.id);
                    outMsg = `New Clan Recruit! Unlocked: ${GOBLIN_TYPES[prize.id].name}!`;
                }
            }

            document.getElementById('spin-reward-text').innerText = outMsg;
            document.getElementById('wheel-close-btn').style.display = 'block';
            saveProfile();
        }

        function drawSpinningWheelContext() {
            let cx = wheelCanvas.width / 2;
            let cy = wheelCanvas.height / 2;
            let radius = cx - 20;
            let arcSize = (Math.PI * 2) / 16;

            wCtx.clearRect(0, 0, wheelCanvas.width, wheelCanvas.height);

            WHEEL_PRIZES.forEach((prize, index) => {
                let startAng = wheelAngle + (index * arcSize);
                let endAng = startAng + arcSize;

                wCtx.fillStyle = prize.color;
                wCtx.beginPath();
                wCtx.moveTo(cx, cy);
                wCtx.arc(cx, cy, radius, startAng, endAng);
                wCtx.closePath();
                wCtx.fill();

                wCtx.strokeStyle = 'rgba(0,0,0,0.2)';
                wCtx.lineWidth = 2;
                wCtx.stroke();

                wCtx.save();
                wCtx.translate(cx, cy);
                wCtx.rotate(startAng + arcSize / 2);
                wCtx.textAlign = "right";
                wCtx.fillStyle = "#FFF";
                wCtx.font = "bold 11px 'Segoe UI', sans-serif";
                wCtx.shadowColor = "black";
                wCtx.shadowBlur = 4;
                wCtx.fillText(prize.text, radius - 15, 4);
                wCtx.restore();
            });

            wCtx.strokeStyle = "#FFD700";
            wCtx.lineWidth = 10;
            wCtx.beginPath();
            wCtx.arc(cx, cy, radius, 0, Math.PI * 2);
            wCtx.stroke();

            for (let i = 0; i < 16; i++) {
                let dotAng = wheelAngle + (i * arcSize);
                let dotX = cx + Math.cos(dotAng) * radius;
                let dotY = cy + Math.sin(dotAng) * radius;

                wCtx.fillStyle = '#FFF';
                wCtx.beginPath();
                wCtx.arc(dotX, dotY, 4, 0, Math.PI * 2);
                wCtx.fill();
            }

            wCtx.fillStyle = '#FFD700';
            wCtx.strokeStyle = '#B8860B';
            wCtx.lineWidth = 4;
            wCtx.beginPath();
            wCtx.arc(cx, cy, 25, 0, Math.PI * 2);
            wCtx.fill();
            wCtx.stroke();

            wCtx.save();
            wCtx.translate(cx, 15);
            if (pinBunkAnimationCounter > 0) {
                wCtx.rotate(Math.sin(pinBunkAnimationCounter * Math.PI * 2) * 0.25);
                pinBunkAnimationCounter -= 0.1; 
            }
            wCtx.fillStyle = '#FFF';
            wCtx.strokeStyle = '#000';
            wCtx.lineWidth = 3;
            wCtx.beginPath();
            wCtx.moveTo(-15, -10);
            wCtx.lineTo(15, -10);
            wCtx.lineTo(0, 20);
            wCtx.closePath();
            wCtx.fill();
            wCtx.stroke();
            wCtx.restore();
        }

        function spawnEnvironment() {
            let viewDistance = (canvas.width * 2) / Math.max(0.15, zoom);
            
            let isMegaTargeting = Date.now() < profile.boosts.megaTargetingEnd;
            let bounceWidthMult = isMegaTargeting ? 2.0 : 1.0;
            
            while (nextSpawnX < cameraX + viewDistance) {
                let rand = Math.random();
                
                if (rand < 0.15) { 
                    environmentObjects.push({ type: 'booster', x: nextSpawnX, y: groundY - 60, w: 150 * bounceWidthMult, h: 60 });
                    nextSpawnX += 450 + Math.random() * 150; 
                } else if (rand < 0.30) { 
                    environmentObjects.push({ type: 'bear', x: nextSpawnX, y: groundY - 70, w: 160 * bounceWidthMult, h: 70 });
                    nextSpawnX += 450 + Math.random() * 150;
                } else if (rand < 0.45) { 
                    environmentObjects.push({ type: 'trampoline', x: nextSpawnX, y: groundY - 40, w: 220 * bounceWidthMult, h: 40 });
                    nextSpawnX += 550 + Math.random() * 200; 
                } else if (rand < 0.50) {
                    let arcWidth = 800; let arcHeight = 450;
                    rainbowObjects.push({ x: nextSpawnX, y: groundY, w: arcWidth, h: arcHeight });
                    environmentObjects.push({ type: 'pot_of_gold', x: nextSpawnX + arcWidth - 80, y: groundY - 70, w: 100, h: 70, collected: false });
                    nextSpawnX += arcWidth + 300;
                } else if (rand < 0.60) {
                    environmentObjects.push({ type: 'obstacle', x: nextSpawnX, y: groundY - 90, w: 120, h: 90 });
                    nextSpawnX += 350 + Math.random() * 100;
                } else {
                    let basePatternHeight = groundY - 150 - Math.random() * 300;
                    let coinDensityModifier = (profile.equippedGoblin === 'rich') ? 2.0 : 1.0;
                    let currencyClusterSize = Math.floor((Math.random() * 6 + 5) * coinDensityModifier); 
                    
                    for(let i=0; i < currencyClusterSize; i++) {
                        let isGem = Math.random() < (Date.now() < profile.boosts.emeraldFrenzyEnd ? 0.20 : 0.03);
                        environmentObjects.push({
                            type: isGem ? 'emerald' : 'coin',
                            x: nextSpawnX + (i * (profile.equippedGoblin === 'rich' ? 45 : 70)), 
                            y: basePatternHeight + (Math.sin(i * 0.8) * 60), 
                            radius: isGem ? 20 : 24, collected: false
                        });
                    }
                    nextSpawnX += (currencyClusterSize * (profile.equippedGoblin === 'rich' ? 45 : 70)) + 150;
                }
            }

            while (nextScenerySpawnX < cameraX + viewDistance) {
                let sceneryTypeChance = Math.random();

                if (sceneryTypeChance < 0.45) {
                    let cloudWidth = 180 + Math.random() * 160;
                    sceneryObjects.push({
                        type: 'cloud', x: nextScenerySpawnX, y: groundY - 350 - Math.random() * 550, 
                        w: cloudWidth, h: cloudWidth * 0.4, layer: Math.random() > 0.5 ? 'fg' : 'bg'
                    });
                    nextScenerySpawnX += Math.random() * 200 + 250;
                } else if (sceneryTypeChance < 0.85) {
                    sceneryObjects.push({
                        type: 'tree', x: nextScenerySpawnX, y: groundY, scale: 0.7 + Math.random() * 0.6,
                        layer: Math.random() > 0.5 ? 'fg' : 'bg'
                    });
                    nextScenerySpawnX += Math.random() * 100 + 80;
                } else {
                    let hillRadius = 400 + Math.random() * 300;
                    sceneryObjects.push({
                        type: 'hill', x: nextScenerySpawnX + hillRadius / 2, y: groundY, radius: hillRadius, layer: 'bg' 
                    });
                    nextScenerySpawnX += hillRadius * 0.6;
                }
            }

            environmentObjects = environmentObjects.filter(obj => obj.x + (obj.w || obj.radius) > cameraX - (canvas.width/zoom));
            rainbowObjects = rainbowObjects.filter(obj => obj.x + obj.w > cameraX - (canvas.width/zoom));
            sceneryObjects = sceneryObjects.filter(obj => {
                let bound = obj.w || obj.radius || 150;
                return obj.x + bound > cameraX - (canvas.width/zoom);
            });
        }

        function handleInput() {
            if (gameState === 'AIMING') {
                gameState = 'LAUNCH_ANIM';
                powerContainer.style.display = 'none';
                instructions.style.display = 'none';
                minimapContainer.style.display = 'block'; 
                
                let upgradeMultiplier = 1 + (profile.upgrades.power - 1) * 0.05;
                if (Date.now() < profile.boosts.doubleSpeedEnd) { upgradeMultiplier *= 2.0; }

                let scaleAdjustmentX = (profile.equippedGoblin === 'baby') ? 1.05 : 1.0;
                let scaleAdjustmentY = (profile.equippedGoblin === 'baby') ? 1.10 : 1.0;

                lockedThrustX = (10 + (power / 100) * 16) * upgradeMultiplier * scaleAdjustmentX;   
                lockedThrustY = (-4 - (power / 100) * 8) * upgradeMultiplier * scaleAdjustmentY;  
            } 
            else if (gameState === 'FLYING') {
                if (goblin.isFartTraveling) return;

                if (profile.equippedGoblin === 'luke' && !goblin.hasFartedBoost) {
                    activateLukeGasBoost();
                    return;
                }

                if (groundY - goblin.y - goblin.radius > 50) {
                    let weightFactor = (profile.equippedGoblin === 'fat') ? 1.5 : 1.0;
                    goblin.vy += 45 * weightFactor; 
                    goblin.isSmashing = true; 
                    
                    instructions.style.display = 'block';
                    instructions.innerText = "SMASH!";
                    setTimeout(() => { if(gameState === 'FLYING') instructions.style.display = 'none'; }, 500);
                }
            }
        }

        function update() {
            let targetCameraX = cameraX;
            let targetCameraY = cameraY; 
            let targetZoom = 1.0;

            processWheelPhysicsTick();

            fartParticles = fartParticles.filter(p => {
                p.x += p.vx; p.y += p.vy; p.alpha -= 0.015; p.radius += 0.4; 
                return p.alpha > 0;
            });

            if (gameState === 'MAIN_MENU' || gameState === 'FLYING') {
                let activeLog = ""; let now = Date.now();
                if (now < profile.boosts.doubleGoldEnd) activeLog += `<span class='boost-txt'>✨ 2x Gold (${Math.ceil((profile.boosts.doubleGoldEnd - now)/1000)}s)</span><br>`;
                if (now < profile.boosts.doubleSpeedEnd) activeLog += `<span class='boost-txt'>⚡ 2x Speed (${Math.ceil((profile.boosts.doubleSpeedEnd - now)/1000)}s)</span><br>`;
                if (now < profile.boosts.doubleGemsEnd) activeLog += `<span class='boost-txt'>💎 2x Emeralds (${Math.ceil((profile.boosts.doubleGemsEnd - now)/1000)}s)</span><br>`;
                if (now < profile.boosts.rubberEarthEnd) activeLog += `<span class='boost-txt'>🌍 Rubber Earth (${Math.ceil((profile.boosts.rubberEarthEnd - now)/1000)}s)</span><br>`;
                if (now < profile.boosts.emeraldFrenzyEnd) activeLog += `<span class='boost-txt'>🤑 Gem Frenzy (${Math.ceil((profile.boosts.emeraldFrenzyEnd - now)/1000)}s)</span><br>`;
                if (now < profile.boosts.megaTargetingEnd) activeLog += `<span class='boost-txt'>🎯 Mega Targets (${Math.ceil((profile.boosts.megaTargetingEnd - now)/1000)}s)</span>`;
                document.getElementById('hud-boosts').innerHTML = activeLog;
            }

            if (gameState === 'MAIN_MENU') {
                if (Math.random() < 0.005) { saveProfile(); } 
                spawnEnvironment();
                cameraX = catapult.x;
                cameraY = catapult.y - 150;
            }
            else if (gameState === 'AIMING') {
                power += powerDirection * POWER_SPEED;
                if (power >= 100) { power = 100; powerDirection = -1; }
                if (power <= 0) { power = 0; powerDirection = 1; }
                powerBar.style.width = power + '%';

                catapult.armAngle = -Math.PI / 2 - (power / 100) * (Math.PI / 2);
                goblin.x = catapult.pivotX + Math.cos(catapult.armAngle) * catapult.armLength;
                goblin.y = catapult.pivotY + Math.sin(catapult.armAngle) * catapult.armLength;
                spawnEnvironment();
            }
            else if (gameState === 'LAUNCH_ANIM') {
                catapult.armAngle += 0.15; 
                goblin.x = catapult.pivotX + Math.cos(catapult.armAngle) * catapult.armLength;
                goblin.y = catapult.pivotY + Math.sin(catapult.armAngle) * catapult.armLength;

                if (catapult.armAngle >= catapult.releaseAngle) {
                    catapult.armAngle = catapult.releaseAngle; 
                    gameState = 'FLYING';
                    goblin.vx = lockedThrustX; goblin.vy = lockedThrustY;
                }
            }
            else if (gameState === 'FLYING') {
                if (goblin.isFartTraveling) {
                    goblin.vx = 60; 
                    goblin.x += goblin.vx;

                    for(let i=0; i<4; i++) {
                        let isBrown = Math.random() < 0.5;
                        fartParticles.push({
                            x: goblin.x - goblin.radius + (Math.random() * 20 - 10),
                            y: goblin.y + (Math.random() * 20 - 10),
                            vx: -15 - Math.random() * 10, 
                            vy: Math.random() * 4 - 2,
                            radius: 12 + Math.random() * 20,
                            alpha: 1.0,
                            color: isBrown ? "#654321" : "#228B22" 
                        });
                    }

                    if (goblin.x >= goblin.fartTargetX) {
                        goblin.x = goblin.fartTargetX;
                        goblin.isFartTraveling = false;
                        goblin.vy = goblin.fartSavedVy; 
                        goblin.vx = 15; 
                    }
                } else {
                    let currentGravity = GRAVITY;
                    
                    let altitudeRatio = Math.max(0, Math.min(1, (groundY - goblin.y) / 3600)); 
                    if (profile.equippedGoblin === 'astronaut' && altitudeRatio > 0.2) {
                        currentGravity *= 0.15; 
                    } else if (profile.equippedGoblin === 'baby') {
                        currentGravity *= 0.80;
                    } else if (profile.equippedGoblin === 'fat') {
                        currentGravity *= 1.35;
                    }

                    goblin.vy += currentGravity; goblin.vx *= AIR_FRICTION; 
                    goblin.x += goblin.vx; goblin.y += goblin.vy;
                }

                targetCameraX = goblin.x;
                
                let heightAboveGround = groundY - goblin.y;
                let maxDesiredHeight = canvas.height * 0.65; 
                let minZoom = 0.45; 
                
                if (heightAboveGround > maxDesiredHeight / minZoom) {
                    targetZoom = minZoom;
                    targetCameraY = goblin.y + (maxDesiredHeight / minZoom);
                } else if (heightAboveGround > maxDesiredHeight) {
                    targetZoom = maxDesiredHeight / heightAboveGround;
                    targetCameraY = groundY; 
                } else {
                    targetZoom = 1.0;
                    targetCameraY = groundY; 
                }

                if (isZoomLocked) {
                    targetZoom = Math.min(zoom, targetZoom);
                }

                spawnEnvironment();

                let hitObject = false;
                for (let i = 0; i < environmentObjects.length; i++) {
                    let obj = environmentObjects[i];
                    if (obj.type === 'coin' || obj.type === 'emerald' || obj.collected || obj.used) continue;

                    if (obj.type === 'obstacle' && profile.equippedGoblin === 'ghost') {
                        continue; 
                    }

                    if (goblin.x + goblin.radius > obj.x && goblin.x - goblin.radius < obj.x + obj.w && goblin.y + goblin.radius > obj.y) { 
                        hitObject = true; goblin.isSmashing = false; 
                        
                        if (goblin.isFartTraveling) {
                            goblin.isFartTraveling = false;
                        }

                        if (obj.type === 'booster' || obj.type === 'pot_of_gold' || obj.type === 'bear') {
                            goblin.y = obj.y - goblin.radius; 
                            let fatForwardBoost = (profile.equippedGoblin === 'fat') ? 8 : 5;
                            goblin.vy = -16; 
                            goblin.vx += fatForwardBoost; 
                            isZoomLocked = true; 
                            
                            if (obj.type === 'pot_of_gold') {
                                obj.collected = true; 
                                // FIXED: Nerfed back to 10%
                                let valueMult = 1 + (profile.upgrades.earnings - 1) * 0.10; 
                                if (Date.now() < profile.boosts.doubleGoldEnd) valueMult *= 2.0;
                                let leprechaunFactor = (profile.equippedGoblin === 'leprechaun') ? 2.0 : 1.0;
                                let treasureYield = Math.floor(20 * valueMult * leprechaunFactor);
                                runCoinsCollected += treasureYield; profile.goldVault += treasureYield;
                            }
                        } else if (obj.type === 'trampoline') {
                            goblin.y = obj.y - goblin.radius;
                            let fatForwardBoost = (profile.equippedGoblin === 'fat') ? 6 : 3;
                            goblin.vy = -24; 
                            goblin.vx += fatForwardBoost; 
                            isZoomLocked = true; 
                        } else if (obj.type === 'obstacle') {
                            goblin.y = obj.y - goblin.radius; goblin.vx *= 0.15; goblin.vy = -6; obj.used = true; 
                            isZoomLocked = true;
                        }
                        break;
                    }
                }

                if (!hitObject && goblin.y + goblin.radius >= groundY) {
                    goblin.y = groundY - goblin.radius; 
                    if (goblin.isFartTraveling) goblin.isFartTraveling = false;

                    // FIXED: Rubber Earth now ends if speed drops too low, preventing an infinite lock
                    if (Date.now() < profile.boosts.rubberEarthEnd && Math.abs(goblin.vx) > 1.5) {
                        goblin.vy = -25; 
                        goblin.vx *= 0.75;
                        isZoomLocked = true;
                    } else {
                        let currentRestitution = RESTITUTION; 
                        if (goblin.isSmashing) { 
                            goblin.vx *= 0.7875; 
                            goblin.isSmashing = false; 
                            currentRestitution = 0.12; 
                        }
                        goblin.vy = -goblin.vy * currentRestitution; 
                        goblin.vx *= GROUND_FRICTION; 
                        if (Math.abs(goblin.vy) < 2) goblin.vy = 0; 
                        isZoomLocked = true; 
                    }
                }

                if (Math.abs(goblin.vx) < 0.1 && goblin.vy === 0 && (goblin.y + goblin.radius >= groundY - 1)) {
                    goblin.vx = 0; gameState = 'GAMEOVER'; triggerGameOver();
                }

                environmentObjects.forEach(obj => {
                    if (!obj.collected) {
                        // FIXED: Nerfed back to 10%
                        let valueMult = 1 + (profile.upgrades.earnings - 1) * 0.10; 
                        if (Date.now() < profile.boosts.doubleGoldEnd) valueMult *= 2.0;

                        if (profile.equippedGoblin === 'magnet' && (obj.type === 'coin' || obj.type === 'emerald')) {
                            let dx = goblin.x - obj.x; let dy = goblin.y - obj.y;
                            let dist = Math.sqrt(dx*dx + dy*dy);
                            if (dist < 300) { 
                                obj.x += (goblin.x - obj.x) * 0.15;
                                obj.y += (goblin.y - obj.y) * 0.15;
                            }
                        }

                        if (obj.type === 'coin' || obj.type === 'emerald') {
                            let dx = goblin.x - obj.x; let dy = goblin.y - obj.y;
                            if (Math.sqrt(dx * dx + dy * dy) < goblin.radius + obj.radius) {
                                obj.collected = true;
                                if (obj.type === 'coin') {
                                    let rewardVal = Math.max(1, Math.floor(1 * valueMult));
                                    runCoinsCollected += rewardVal; profile.goldVault += rewardVal;
                                } else if (obj.type === 'emerald') {
                                    let gemYield = (Date.now() < profile.boosts.doubleGemsEnd) ? 2 : 1;
                                    runGemsCollected += gemYield; profile.gemVault += gemYield;
                                }
                            }
                        } 
                    }
                });

                let currentDistance = Math.max(0, (goblin.x / 60)); 
                if (currentDistance > distanceTravelled) {
                    distanceTravelled = currentDistance;
                }

                let activeZoneIdx = 0;
                for (let i = 0; i < CHECKPOINTS.length; i++) {
                    if (distanceTravelled >= CHECKPOINTS[i].dist) {
                        activeZoneIdx = i;
                    }
                }
                
                if (CHECKPOINTS[activeZoneIdx] !== currentBiome) {
                    currentBiome = CHECKPOINTS[activeZoneIdx];
                    showZoneNotification("ZONE UNLOCKED:\n" + currentBiome.name + "!");
                    
                    if (activeZoneIdx > profile.highestCheckpoint) {
                        profile.highestCheckpoint = activeZoneIdx;
                        saveProfile();
                    }
                }

                let nextZoneIdx = activeZoneIdx + 1;
                if (nextZoneIdx < CHECKPOINTS.length) {
                    let startD = CHECKPOINTS[activeZoneIdx].dist;
                    let endD = CHECKPOINTS[nextZoneIdx].dist;
                    let pct = ((distanceTravelled - startD) / (endD - startD)) * 100;
                    
                    minimapStart.innerText = startD + "m";
                    minimapEnd.innerText = endD + "m";
                    minimapZoneName.innerText = CHECKPOINTS[activeZoneIdx].name;
                    minimapFill.style.width = Math.min(100, Math.max(0, pct)) + "%";
                } else {
                    minimapStart.innerText = CHECKPOINTS[activeZoneIdx].dist + "m";
                    minimapEnd.innerText = "INFINITY";
                    minimapZoneName.innerText = currentBiome.name;
                    minimapFill.style.width = "100%";
                }

                let currentHeight = Math.max(0, (groundY - goblin.y - goblin.radius) / 60);
                maxHeight = Math.max(maxHeight, currentHeight);

                updateHUD();
            }

            cameraX = targetCameraX; 
            cameraY += (targetCameraY - cameraY) * 0.15; 
            zoom += (targetZoom - zoom) * 0.1; 
        }

                            ctx.save(); ctx.globalAlpha = 0.6; ctx.strokeStyle = '#87CEEB'; ctx.lineWidth = 2;
                            ctx.beginPath(); ctx.arc(obj.x, obj.y, obj.h * 0.2, 0, Math.PI * 2); ctx.stroke();
                            ctx.beginPath(); ctx.arc(obj.x + 30, obj.y - 40, obj.h * 0.1, 0, Math.PI * 2); ctx.stroke();
                            ctx.beginPath(); ctx.arc(obj.x - 20, obj.y - 20, obj.h * 0.15, 0, Math.PI * 2); ctx.stroke();
                            ctx.restore();
                        } else {
                            ctx.save(); ctx.globalAlpha = 0.55; ctx.fillStyle = '#FFFFFF';
                            let r = obj.h * 0.5;
                            ctx.beginPath();
                            ctx.arc(obj.x, obj.y, r, 0, Math.PI * 2);
                            ctx.arc(obj.x + r * 1.2, obj.y - r * 0.3, r * 1.3, 0, Math.PI * 2);
                            ctx.arc(obj.x + r * 2.6, obj.y, r * 0.9, 0, Math.PI * 2);
                            ctx.fillRect(obj.x, obj.y - r * 0.3, r * 2.6, r * 1.3);
                            ctx.fill(); ctx.restore();
                        }
                    }
                }
            });

            ctx.restore();
        }

        let lastFrameTime = 0;
        const targetFPS = 60;
        const frameDelay = 1000 / targetFPS;

        function loop(timestamp) {
            requestAnimationFrame(loop);
            let deltaTime = timestamp - lastFrameTime;
            
            if (deltaTime >= frameDelay) {
                lastFrameTime = timestamp - (deltaTime % frameDelay);
                update();
                draw();
            }
        }

        resize();
        processOfflineEarnings();
        resetToMenuState();
        requestAnimationFrame(loop);
